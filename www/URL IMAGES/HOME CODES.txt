      function visibleButtons(){return previewButtons.filter((button)=>{const card=button.closest('.project-card');return card && !card.classList.contains('hidden');});}

      function closeViewer(){viewer.classList.remove('active');viewer.setAttribute('aria-hidden','true');viewerImage.src='';document.body.style.overflow='';}
      previewButtons.forEach((button,idx)=>button.addEventListener('click',()=>{const pool=visibleButtons();const local=pool.indexOf(button);openByIndex(local>-1?local:idx);}));
      viewerPrev.addEventListener('click',()=>openByIndex(activeIndex-1));
      viewerNext.addEventListener('click',()=>openByIndex(activeIndex+1));
      viewerClose.addEventListener('click',closeViewer);

      filterButtons.forEach((button)=>{button.addEventListener('click',()=>{filterButtons.forEach((item)=>{item.classList.remove('active');item.setAttribute('aria-pressed','false');});button.classList.add('active');button.setAttribute('aria-pressed','true');const filter=button.dataset.filter||'all';cards.forEach((card)=>{const show=filter==='all' || card.dataset.category===filter;card.classList.toggle('hidden',!show);});filterLabel.textContent=filter==='all'?'Showing all projects':'Showing '+filter.toUpperCase()+' projects';if(viewer.classList.contains('active'))closeViewer();});});

      document.addEventListener('keydown',(event)=>{if(event.key==='Escape'){closeMenu();if(viewer.classList.contains('active'))closeViewer();}if(!viewer.classList.contains('active'))return;if(event.key==='ArrowLeft')openByIndex(activeIndex-1);if(event.key==='ArrowRight')openByIndex(activeIndex+1);});

      const revealItems=[...document.querySelectorAll('.reveal')];

          document.body.classList.remove('menu-open');

          document.body.classList.remove('menu-open');
